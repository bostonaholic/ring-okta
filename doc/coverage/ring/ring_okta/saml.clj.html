001  (ns ring.ring-okta.saml
002    (:require [clojure.data.codec.base64 :as b64]
003              [clojure.java.io :as io]
004              [clojure.string :as string])
005    (:import (com.okta.saml SAMLValidator)))
006  
007  (defn- get-valid-user-id [okta-config saml-response]
008    (let [validator (SAMLValidator.)
009          config (.getConfiguration validator okta-config)
010          decoded-saml-response (String. (b64/decode (.getBytes saml-response "UTF-8")))
011          valid-saml-response (.getSAMLResponse validator decoded-saml-response config)]
012      (.getUserID valid-saml-response)))
013  
014  (defn respond-to-okta-post [params okta-config-location]
015    (let [okta-config (slurp (io/reader okta-config-location))]
016      {:redirect-url (:RelayState params)
017       :authenticated-user-email (string/lower-case
018                                  (get-valid-user-id okta-config (:SAMLResponse params)))}))
